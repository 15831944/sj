-- tasks --
* retain class values
* if function is heap and has this value then all arguments must be heap

-- operators --
	a.x 		-> getX()
	a.x = y 	-> setX(y)
	a * b 		-> a.multiply(b)
	a / b 		-> a.divide(b)
	a + b 		-> a.add(b)
	a - b 		-> a.subtract(b)
	a *= b 		-> a = a.add(b)
	a /= b 		-> a = a.divide(b)
	a += b 		-> a = a.add(b)
	a -= b 		-> a = a.subtract(b)
	a as type	-> a.asType()'t

-- tuple --
* create interface for all array classes
* create special class for 1, 2, 3 element arrays
* special array assign operation
	[a, b] = [1, 2]
	t: tuple![i32, i32](1, 2)
	a = t.getAt(0)
	b = t.getAt(1)
