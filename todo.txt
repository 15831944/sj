-- operators --
	a.x 		-> getX()
	a.x = y 	-> setX(y)
	a * b 		-> a.multiply(b)
	a / b 		-> a.divide(b)
	a + b 		-> a.add(b)
	a - b 		-> a.subtract(b)
	a *= b 		-> a = a.add(b)
	a /= b 		-> a = a.divide(b)
	a += b 		-> a = a.add(b)
	a -= b 		-> a = a.subtract(b)
	a < b 		-> a.isLess(b)
	a <= b 		-> a.isLessOrEqual(b)
	a > b 		-> a.isGreater(b)
	a >= b 		-> a.isGreaterOrEqual(b)
	a == b 		-> a.isEqual(b)
	a as type	-> a.asType()'t
