-- interface --
* cast from interface to another interface
* call interace method

void* foo_asInterface(sjs_object* _parent, int sji_foo_id) {
	switch (sji_foo_id) {
	case sji_foo_id:
		return foo_asFoo(_parent);
	}
	return nullptr;
}

-- tuple --
* create interface for all array classes
* create special class for 1, 2, 3 element arrays
* special array assign operation
	[a, b] = [1, 2]
	t: tuple![i32, i32](1, 2)
	a = t.getAt(0)
	b = t.getAt(1)
